#include <stdio.h>

int my_strlen(char * str){
	int len = 0;
	while (*str++){
		len++;
	}
	return len;
}

int my_strlen2(char * str){
	int len = 0;
	while (str[len++]);
	return len-1;
}

void my_strcpy(char * src, char * dest){
	while (*dest++ = *src++);
}

int my_strcmp(char * s1, char * s2){
	int idx = 0;
	while (s1[idx]){
		if (s1[idx] != s2[idx]){
			break;
		}
		idx++;
	}
	if (s1[idx] > s2[idx]){
		return 1;
	}
	else if (s1[idx] < s2[idx]){
		return -1;
	}
	return 0;
}

int my_atoi(char * sNum){//123
	char ch;
	int val = 0;
	int digit;
	int sign = 1;

		sign = -1;
		sNum++;
	}
	while(ch = *sNum++){

		//		if (ch < '0' || ch > '9'){break;}
		digit = ch - '0';

		val = val * 10 + digit;

	return val*sign;
}

void my_itoa(int iNum, char * sNum){//123
	char ch;
	int idx = 0;
	while (iNum){
		ch = iNum % 10 +'0';
		sNum[idx++] = ch;
		iNum = iNum / 10;
	}
	sNum[idx] = 0;
	int len = idx / 2;

	for (int i = 0; i < len; i++){
		int t = sNum[i];
		sNum[i]  = sNum[idx - 1-i];
		sNum[idx - 1-i] = t;
	}
}
